\begin{abstract}
We introduce the observable defect operator $D_\Pi = (I - P_\Pi) U_T P_\Pi$ for a finite dynamical system $(X, T)$ and a partition $\Pi$ of $X$. The defect operator measures the component of one-step observable evolution that escapes the observable representation. We prove that $D_\Pi = 0$ if and only if the observable subspace $V_\Pi$ is Koopman-invariant, equivalently if and only if the quotient dynamics on $X/\Pi$ is well-defined. This theorem unifies finite dynamical systems, Koopman operator theory, and quotient system construction. We verify all claims through exhaustive computational enumeration and formal proof in Lean~4. Negative results---including disproved conjectures and spectral collisions---are reported explicitly. The 4-digit Kaprekar routine serves as a benchmark: its 55-class Nerode partition is an exact quotient with nilpotency index~6.
\end{abstract}

\section{Mathematical Framework}

\begin{definition}[Finite Deterministic Dynamical System]
A pair $(X, T)$ where $X$ is a finite set and $T: X \to X$ is an arbitrary function.
\end{definition}

\begin{definition}[Observable Space]
$\mathcal{F}(X) = \{f: X \to \mathbb{R}\}$, the space of real-valued functions on $X$, with inner product $\langle f, g \rangle = \sum_{x \in X} f(x)g(x)$.
\end{definition}

\begin{definition}[Koopman Pullback Operator]
$U_T: \mathcal{F}(X) \to \mathcal{F}(X)$ defined by $(U_T f)(x) = f(T(x))$.
\end{definition}

\begin{definition}[Observable Subspace]
For a partition $\Pi$ of $X$, $V_\Pi = \{f \in \mathcal{F}(X) : x \sim_\Pi y \Rightarrow f(x) = f(y)\}$.
\end{definition}

\begin{definition}[Observable Projection]
$P_\Pi: \mathcal{F}(X) \to V_\Pi$ is the orthogonal projection onto $V_\Pi$.
\end{definition}

\begin{definition}[Observable Defect Operator]
\[
D_\Pi = (I - P_\Pi) U_T P_\Pi.
\]
\end{definition}

\section{Defect Operator Theorem}

\begin{theorem}[Fundamental Decomposition]
$U_T P_\Pi = P_\Pi U_T P_\Pi + D_\Pi$.
\end{theorem}

\begin{proof}
Direct from Definition~5.1.
\end{proof}

\begin{theorem}[Defect Vanishing]
\[
D_\Pi = 0 \iff U_T(V_\Pi) \subseteq V_\Pi \iff (x \sim_\Pi y \Rightarrow T(x) \sim_\Pi T(y)).
\]
\end{theorem}

\begin{proof}
$(\Rightarrow)$ Assume $D_\Pi = 0$. Then $(I - P_\Pi) U_T P_\Pi = 0$, so $U_T P_\Pi = P_\Pi U_T P_\Pi$. For any $f \in V_\Pi$, $U_T f = P_\Pi U_T f \in V_\Pi$. Thus $U_T(V_\Pi) \subseteq V_\Pi$.

$(\Leftarrow)$ Assume $x \sim_\Pi y \Rightarrow T(x) \sim_\Pi T(y)$. Then for any $f \in V_\Pi$, $U_T f$ is constant on blocks of $\Pi$, so $U_T f \in V_\Pi$. Thus $U_T(V_\Pi) \subseteq V_\Pi$, meaning $(I - P_\Pi) U_T P_\Pi = 0$.
\end{proof}

\begin{theorem}[Exact Quotient]
$D_\Pi = 0 \iff$ the quotient dynamics $T_\Pi: X/\Pi \to X/\Pi$ is well-defined by $T_\Pi([x]) = [T(x)]$.
\end{theorem}

\begin{proof}
By Theorem~3.2, $D_\Pi = 0 \iff T$ respects partition blocks $\iff [T(x)]$ depends only on $[x]$.
\end{proof}
