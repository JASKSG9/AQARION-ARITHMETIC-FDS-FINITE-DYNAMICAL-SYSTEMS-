\section{Mathematical Framework}

\begin{definition}[Finite Deterministic Dynamical System]
A pair $(X, T)$ where $X$ is a finite set and $T: X \to X$ is an arbitrary function.
\end{definition}

\begin{definition}[Observable Space]
$\mathcal{F}(X) = \{f: X \to \mathbb{R}\}$, the space of real-valued functions on $X$, with inner product $\langle f, g \rangle = \sum_{x \in X} f(x)g(x)$.
\end{definition}

\begin{definition}[Koopman Pullback Operator]
$U_T: \mathcal{F}(X) \to \mathcal{F}(X)$ defined by $(U_T f)(x) = f(T(x))$.
\end{definition}

\begin{definition}[Observable Subspace]
For a partition $\Pi$ of $X$, $V_\Pi = \{f \in \mathcal{F}(X) : x \sim_\Pi y \Rightarrow f(x) = f(y)\}$.
\end{definition}

\begin{definition}[Observable Projection]
$P_\Pi: \mathcal{F}(X) \to V_\Pi$ is the orthogonal projection onto $V_\Pi$.
\end{definition}

\begin{definition}[Observable Defect Operator]
\[
D_\Pi = (I - P_\Pi) U_T P_\Pi.
\]
\end{definition}

\section{Defect Operator Theorem}

\begin{theorem}[Fundamental Decomposition]
$U_T P_\Pi = P_\Pi U_T P_\Pi + D_\Pi$.
\end{theorem}

\begin{proof}
Direct from Definition~5.1.
\end{proof}

\begin{theorem}[Defect Vanishing]
\[
D_\Pi = 0 \iff U_T(V_\Pi) \subseteq V_\Pi \iff (x \sim_\Pi y \Rightarrow T(x) \sim_\Pi T(y)).
\]
\end{theorem}

\begin{proof}
$(\Rightarrow)$ Assume $D_\Pi = 0$. Then $(I - P_\Pi) U_T P_\Pi = 0$, so $U_T P_\Pi = P_\Pi U_T P_\Pi$. For any $f \in V_\Pi$, $U_T f = P_\Pi U_T f \in V_\Pi$. Thus $U_T(V_\Pi) \subseteq V_\Pi$.

$(\Leftarrow)$ Assume $x \sim_\Pi y \Rightarrow T(x) \sim_\Pi T(y)$. Then for any $f \in V_\Pi$, $U_T f$ is constant on blocks of $\Pi$, so $U_T f \in V_\Pi$. Thus $U_T(V_\Pi) \subseteq V_\Pi$, meaning $(I - P_\Pi) U_T P_\Pi = 0$.
\end{proof}

\begin{theorem}[Exact Quotient]
$D_\Pi = 0 \iff$ the quotient dynamics $T_\Pi: X/\Pi \to X/\Pi$ is well-defined by $T_\Pi([x]) = [T(x)]$.
\end{theorem}

\begin{proof}
By Theorem~3.2, $D_\Pi = 0 \iff T$ respects partition blocks $\iff [T(x)]$ depends only on $[x]$.
\end{proof}

\section{Introduction}

\section{Preliminaries}

Let $X$ be a finite set and $T:X\to X$ an arbitrary map. Let $\mathcal F(X)=\{f:X\to\mathbb R\}$ with inner product $\langle f,g\rangle=\sum_{x\in X}f(x)g(x)$. The \emph{Koopman operator} $U_T:\mathcal F(X)\to\mathcal F(X)$ is defined by $(U_T f)(x)=f(T(x))$.

For a partition $\Pi$ of $X$, the \emph{observable subspace} is
\[
V_\Pi \;=\; \{f\in\mathcal F(X): x\sim_\Pi y \Rightarrow f(x)=f(y)\}.
\]
Let $P_\Pi:\mathcal F(X)\to V_\Pi$ be the orthogonal projection.

\begin{definition}[Observable Defect Operator]\label{def:defect}
The \emph{observable defect operator} is
\[
D_\Pi \;:=\; (I-P_\Pi)\,U_T\,P_\Pi.
\]
\end{definition}

The operator $D_\Pi$ maps $V_\Pi$ to $V_\Pi^\perp$; it measures the component of one-step observable evolution that escapes the observable representation.

\section{Main Results}

\begin{theorem}[Fundamental Decomposition]\label{thm:decomp}
$U_T P_\Pi = P_\Pi U_T P_\Pi + D_\Pi$.
\end{theorem}
\begin{proof}
Immediate from Definition~\ref{def:defect}.
\end{proof}

\begin{theorem}[Defect Vanishing]\label{thm:vanish}
The following are equivalent:
\begin{enumerate}
    \item[(i)] $D_\Pi = 0$.
    \item[(ii)] $U_T(V_\Pi)\subseteq V_\Pi$.
    \item[(iii)] $x\sim_\Pi y \Rightarrow T(x)\sim_\Pi T(y)$.
\end{enumerate}
\end{theorem}
\begin{proof}
(i)$\Rightarrow$(ii): If $D_\Pi=0$, then $(I-P_\Pi)U_T P_\Pi=0$, so $U_T P_\Pi = P_\Pi U_T P_\Pi$. For any $f\in V_\Pi$, $U_T f = P_\Pi U_T f \in V_\Pi$.

(ii)$\Rightarrow$(iii): If $U_T(V_\Pi)\subseteq V_\Pi$, then for any block $B\in\Pi$, the indicator $\mathbf 1_B\in V_\Pi$, so $U_T\mathbf 1_B = \mathbf 1_{T^{-1}(B)}\in V_\Pi$. Thus $T^{-1}(B)$ is a union of blocks, equivalently $x\sim y\Rightarrow T(x)\sim T(y)$.

(iii)$\Rightarrow$(i): If $T$ respects blocks, then for any $f\in V_\Pi$, $U_T f$ is constant on blocks, so $U_T f\in V_\Pi$. Hence $(I-P_\Pi)U_T P_\Pi=0$.
\end{proof}

\begin{theorem}[Exact Quotient]\label{thm:quotient}
$D_\Pi=0$ if and only if the quotient map $T_\Pi:X/\Pi\to X/\Pi$ given by $T_\Pi([x])=[T(x)]$ is well-defined.
\end{theorem}
\begin{proof}
By Theorem~\ref{thm:vanish}, $D_\Pi=0$ iff $T$ respects partition blocks, which is exactly the condition for $[T(x)]$ to depend only on $[x]$.
\end{proof}
