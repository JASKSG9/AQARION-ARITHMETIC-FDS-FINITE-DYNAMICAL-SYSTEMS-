def mat_zero(n: int, m: int) -> List[List[Fraction]]:

return [[Fraction(0) for _ in range(m)] for _ in range(n)]


def mat_identity(n: int) -> List[List[Fraction]]:

return [[Fraction(1) if i == j else Fraction(0) for j in range(n)] for i in range(n)]


def mat_mult(A: List[List[Fraction]], B: List[List[Fraction]]) -> List[List[Fraction]]:

n = len(A)

m = len(B[0])

p = len(B)

C = mat_zero(n, m)

for i in range(n):

for j in range(m):

s = Fraction(0)

for k in range(p):

s += A[i][k] * B[k][j]

C[i][j] = s

return C


def mat_sub(A: List[List[Fraction]], B: List[List[Fraction]]) -> List[List[Fraction]]:

return [[A[i][j] - B[i][j] for j in range(len(A[0]))] for i in range(len(A))]


def mat_is_zero(M: List[List[Fraction]]) -> bool:

return all(x == Fraction(0) for row in M for x in row)


def mat_frobenius_norm_sq(M: List[List[Fraction]]) -> Fraction:

return sum(x * x for row in M for x in row)

class FDDS:

def init(self, transition: Dict[Any, Any]):

self.f = {}

for k, v in transition.items():

try:

kk = int(k)

except (ValueError, TypeError):

kk = k

try:

vv = int(v)

except (ValueError, TypeError):

vv = v

self.f[kk] = vv


    self.states = sorted(self.f.keys(), key=lambda x: (isinstance(x, str), x))    
    self.n = len(self.states)    
    self._index = {s: i for i, s in enumerate(self.states)}    
    
def orbit(self, x, max_steps: int = 10000) -> Tuple[List[Any], List[Any]]:    
    seen = {}    
    path = []    
    current = x    
    step = 0    
    while current not in seen and step < max_steps:    
        seen[current] = len(path)    
        path.append(current)    
        current = self.f[current]    
        step += 1    
    if current in seen:    
        return path[:seen[current]], path[seen[current]:]    
    return path, []    
    
def transition_hash(self) -> str:    
    canonical = json.dumps(self.f, sort_keys=True, separators=(',', ':'))    
    return hashlib.sha256(canonical.encode()).hexdigest()    
    
def to_dict(self) -> Dict:    
    return {    
        "states": self.states,    
        "transition": {str(k): v for k, v in self.f.items()},    
        "size": self.n,    
        "transition_hash": self.transition_hash()    
    }

def find_cycles(system: FDDS) -> List[List[Any]]:

visited = set()

cycles = []

for start in system.states:

if start in visited:

continue

path = []

seen = {}

x = start

while x not in seen:

if x in visited and x not in seen:

break

seen[x] = len(path)

path.append(x)

x = system.f[x]

if x in seen:

cycle = path[seen[x]:]

cycles.append(cycle)

visited.update(path)

else:

visited.update(path)

return cycles


def image_filtration(system: FDDS, depth: int = 12) -> List[int]:

current = set(system.states)

result = []

for _ in range(depth):

result.append(len(current))

if not current:

break

current = {system.f[x] for x in current}

return result


def compute_observable_partition(system: FDDS) -> List[List[Any]]:

groups: Dict[tuple, List[Any]] = {}

for state in system.states:

transient, cycle = system.orbit(state)

key = (len(transient), tuple(sorted(cycle)) if cycle else None)

groups.setdefault(key, []).append(state)

return list(groups.values())

def koopman_matrix(system: FDDS) -> List[List[Fraction]]:

n = system.n

K = mat_zero(n, n)

for x in system.states:

i = system._index[x]

j = system._index[system.f[x]]

K[j][i] = Fraction(1)

return K


def projection_matrix(partition: List[List[Any]], states: List[Any]) -> List[List[Fraction]]:

n = len(states)

index = {s: i for i, s in enumerate(states)}

P = mat_zero(n, n)

for block in partition:

m = len(block)

frac = Fraction(1, m)

for a in block:

for b in block:

P[index[a]][index[b]] = frac

return P


def defect_operator(system: FDDS, partition: List[List[Any]]) -> List[List[Fraction]]:

K = koopman_matrix(system)

P = projection_matrix(partition, system.states)

I = mat_identity(system.n)

I_minus_P = mat_sub(I, P)

return mat_mult(mat_mult(I_minus_P, K), P)

def kaprekar_step(n: int, base: int = 10, digits: int = 4) -> int:

s = str(n).zfill(digits)

desc = int("".join(sorted(s, reverse=True)))

asc = int("".join(sorted(s)))

return desc - asc


def kaprekar_system(digits: int = 4, base: int = 10) -> FDDS:

max_val = base ** digits

transition = {}

for i in range(max_val):

transition[i] = kaprekar_step(i, base, digits)

return FDDS(transition)

return {

class EvidenceObject:

"""

A certificate that answers one explicit claim.

Mathematical claims only. No derived measurements in hash.

"""


def __init__(self, claim_id: str, system: FDDS, metadata: Dict = None):    
    if claim_id not in CLAIM_REGISTRY:    
        raise ValueError(f"Unknown claim: {claim_id}")    
    self.claim_id = claim_id    
    self.system = system    
    self.metadata = metadata or {}    
    self.provenance = get_provenance()    
    self.evidence: Dict[str, Any] = {}    
    
def compute_claim_001(self):    
    """Observable partition P is exact (D = 0)."""    
    partition = compute_observable_partition(self.system)    
    D = defect_operator(self.system, partition)    
    self.evidence["partition"] = partition    
    self.evidence["defect_is_zero"] = mat_is_zero(D)    
    self.evidence["defect_norm_sq"] = str(mat_frobenius_norm_sq(D))    
    return self.evidence["defect_is_zero"]    
    
def compute_claim_002(self):    
    """Cycle enumeration is complete."""    
    cycles = find_cycles(self.system)    
    self.evidence["cycles"] = cycles    
    self.evidence["cycle_count"] = len(cycles)    
    self.evidence["cycle_lengths"] = [len(c) for c in cycles]    
    return True  # Exhaustive for finite systems    
    
def compute_claim_003(self):    
    """Image filtration stabilizes at |C|."""    
    filt = image_filtration(self.system)    
    self.evidence["filtration"] = filt    
    cycles = find_cycles(self.system)    
    periodic = sum(len(c) for c in cycles)    
    self.evidence["stabilizes_at"] = filt[-1] if filt else 0    
    self.evidence["expected_periodic"] = periodic    
    self.evidence["stabilization_correct"] = filt[-1] == periodic if filt else False    
    return self.evidence["stabilization_correct"]    
    
def compute_claim_004(self):    
    """Transition hash matches."""    
    self.evidence["transition_hash"] = self.system.transition_hash()    
    return True    
    
def build(self) -> Dict[str, Any]:    
    # Compute all claims    
    self.compute_claim_001()    
    self.compute_claim_002()    
    self.compute_claim_003()    
    self.compute_claim_004()    
        
    # Mathematical claims (immutable, hashed)    
    mathematical_claims = {    
        "claim_id": self.claim_id,    
        "claim_statement": CLAIM_REGISTRY[self.claim_id]["statement"],    
        "system": self.system.to_dict(),    
        "evidence": {    
            k: v for k, v in self.evidence.items()    
            if k not in ["computation_time", "memory_usage"]  # Exclude derived    
        }    
    }    
        
    # Derived measurements (audit only, not hashed)    
    derived = {    
        "provenance": self.provenance,    
        "metadata": self.metadata    
    }    
        
    # Hash only mathematical claims    
    canonical = json.dumps(mathematical_claims, sort_keys=True, separators=(',', ':'))    
    certificate_hash = hashlib.sha256(canonical.encode()).hexdigest()    
        
    return {    
        "aqarion_version": "0.4.0-final",    
        "certificate_version": "2.1.0",    
        **mathematical_claims,    
        **derived,    
        "certificate_hash": certificate_hash    
    }    
    
def save(self, path: str) -> str:    
    cert = self.build()    
    with open(path, 'w') as f:    
        json.dump(cert, f, indent=2)    
    return cert["certificate_hash"]

def verify_evidence(cert: Dict[str, Any]) -> Dict[str, Any]:

"""

Independent recomputation of all mathematical claims.

Does NOT trust stored values. Recomputes from system definition.

"""

report = {

"claim_id": cert.get("claim_id", "UNKNOWN"),

"overall_valid": False,

"checks": {}

}


# 1. Hash integrity (mathematical claims only)    
stored_hash = cert.get("certificate_hash", "")    
hash_payload = {    
    "claim_id": cert.get("claim_id"),    
    "claim_statement": cert.get("claim_statement"),    
    "system": cert.get("system"),    
    "evidence": cert.get("evidence")    
}    
canonical = json.dumps(hash_payload, sort_keys=True, separators=(',', ':'))    
computed_hash = hashlib.sha256(canonical.encode()).hexdigest()    
report["checks"]["hash_integrity"] = {    
    "valid": computed_hash == stored_hash,    
    "stored": stored_hash[:16] + "...",    
    "computed": computed_hash[:16] + "..."    
}    
    
# 2. System reconstruction    
sys_data = cert.get("system", {})    
if not sys_data or "transition" not in sys_data:    
    report["checks"]["system_reconstruction"] = {"valid": False, "error": "missing system data"}    
    return report    
    
transition = {str(k): v for k, v in sys_data["transition"].items()}    
system = FDDS(transition)    
report["checks"]["system_reconstruction"] = {    
    "valid": True,    
    "states": system.n,    
    "transition_hash_matches": system.transition_hash() == sys_data.get("transition_hash", "")    
}    
    
# 3. Claim 001: Defect is zero    
partition = compute_observable_partition(system)    
D = defect_operator(system, partition)    
d_zero = mat_is_zero(D)    
claimed_d_zero = cert.get("evidence", {}).get("defect_is_zero")    
report["checks"]["claim_001_defect_zero"] = {    
    "valid": d_zero == claimed_d_zero,    
    "recomputed": d_zero,    
    "claimed": claimed_d_zero    
}    
    
# 4. Claim 002: Cycles    
cycles = find_cycles(system)    
claimed_cycles = cert.get("evidence", {}).get("cycles", [])    
cycles_ok = len(cycles) == len(claimed_cycles)    
if cycles_ok:    
    cycle_lens = sorted(len(c) for c in cycles)    
    claimed_lens = sorted(len(c) for c in claimed_cycles)    
    cycles_ok = cycle_lens == claimed_lens    
report["checks"]["claim_002_cycles"] = {    
    "valid": cycles_ok,    
    "recomputed_count": len(cycles),    
    "claimed_count": len(claimed_cycles)    
}    
    
# 5. Claim 003: Filtration    
filt = image_filtration(system)    
claimed_filt = cert.get("evidence", {}).get("filtration", [])    
report["checks"]["claim_003_filtration"] = {    
    "valid": filt == claimed_filt,    
    "recomputed": filt,    
    "claimed": claimed_filt    
}    
    
# Overall    
report["overall_valid"] = all(    
    c.get("valid", False) for c in report["checks"].values()    
)    
return report

```markdown
# AQARION Mathematical Definitions

Status: Frozen for v0.4.1

---

## AQ-DEF-001: Finite Deterministic Dynamical System (FDDS)

A **finite deterministic dynamical system** is a pair:

$$(X, f)$$

where:
- $X$ is a finite set (the **state space**)
- $f: X \to X$ is a function (the **transition map**)

The system is **deterministic** because $f(x)$ is uniquely defined for every $x \in X$.

---

## AQ-DEF-002: Orbit

For $x \in X$, the **orbit** of $x$ is the sequence:

$$x, f(x), f^2(x), f^3(x), \ldots$$

where $f^k$ denotes $k$-fold composition.

---

## AQ-DEF-003: Transient and Periodic States

For $x \in X$:

- The **transient part** of $x$ is the initial segment of the orbit before entering a cycle.
- The **periodic part** (or **cycle**) of $x$ is the repeating segment.

Formally, there exist integers $t \geq 0$ and $c \geq 1$ such that:

$$f^{t+c}(x) = f^t(x)$$

and $c$ is minimal with this property.

---

## AQ-DEF-004: Image Filtration

For $(X, f)$, the **image filtration** is:

$$\operatorname{Im}(f^0) \supseteq \operatorname{Im}(f^1) \supseteq \operatorname{Im}(f^2) \supseteq \cdots$$

where $\operatorname{Im}(f^k) = \{f^k(x) : x \in X\}$.

---

## AQ-DEF-005: Koopman Operator

For $(X, f)$ with $|X| = n$, the **Koopman operator** $K$ is the $n \times n$ matrix:

$$K_{ij} = \begin{cases} 1 & \text{if } f(x_j) = x_i \\ 0 & \text{otherwise} \end{cases}$$

---

## AQ-DEF-006: Observable Partition

A **partition** $\Pi$ of $X$ is a set of disjoint blocks whose union is $X$.

A partition is **observable** (or **Koopman-invariant**) if:

$$K(\text{span}\{\mathbf{1}_B : B \in \Pi\}) \subseteq \text{span}\{\mathbf{1}_B : B \in \Pi\}$$

---

## AQ-DEF-007: Projection Operator

For partition $\Pi$, the **projection** $P_\Pi$ onto $\Pi$-constant vectors is:

$$(P_\Pi)_{ab} = \begin{cases} \frac{1}{|B|} & \text{if } a, b \in B \text{ for some } B \in \Pi \\ 0 & \text{otherwise} \end{cases}$$

---

## AQ-DEF-008: Defect Operator

For partition $\Pi$, the **defect operator** is:

$$D_\Pi = (I - P_\Pi) K P_\Pi$$

The partition is Koopman-invariant if and only if $D_\Pi = 0$.
```

```markdown
# AQARION Theorems

Status: Frozen for v0.4.1

---

## AQ-THM-001: Finite Orbit Decomposition

**Statement:** For every finite deterministic system $(X, f)$:

$$X = T \cup C$$

where $T$ is the set of transient states and $C$ is the set of periodic states.

**Proof:** By the pigeonhole principle, every orbit must eventually repeat. The first repeated state closes a cycle. All states before the first repeated state are transient.

**Status:** Proven. Classical result.

---

## AQ-THM-002: Image Filtration Stabilization

**Statement:** For finite $(X, f)$:

$$\operatorname{Im}(f) \supseteq \operatorname{Im}(f^2) \supseteq \operatorname{Im}(f^3) \supseteq \cdots$$

stabilizes at exactly $C$, the set of periodic states:

$$\bigcap_{k=1}^{\infty} \operatorname{Im}(f^k) = C$$

**Proof:** 
- (Monotonicity) $\operatorname{Im}(f^{k+1}) \subseteq \operatorname{Im}(f^k)$ since $f^{k+1}(X) = f(f^k(X)) \subseteq f^k(X)$.
- (Stabilization) The sequence is a descending chain of finite sets, so it stabilizes.
- (Limit) A state is in the limit iff it has arbitrarily long preimages, which is equivalent to being on a cycle.

**Status:** Proven. Classical result.

---

## AQ-THM-003: Defect Zero Criterion

**Statement:** For a partition $\Pi$ of $X$:

$$D_\Pi = 0 \iff \Pi \text{ is Koopman-invariant}$$

**Proof:**
- ($\Rightarrow$) If $D_\Pi = (I - P_\Pi) K P_\Pi = 0$, then $K P_\Pi = P_\Pi K P_\Pi$. For any $\Pi$-constant vector $v$, $P_\Pi v = v$, so $K v = P_\Pi K v$, meaning $K v$ is $\Pi$-constant. Thus $\Pi$ is invariant.
- ($\Leftarrow$) If $\Pi$ is invariant, then $K$ maps $\Pi$-constant vectors to $\Pi$-constant vectors. For any $v$, $P_\Pi v$ is $\Pi$-constant, so $K P_\Pi v$ is $\Pi$-constant, so $P_\Pi K P_\Pi v = K P_\Pi v$. Thus $(I - P_\Pi) K P_\Pi v = 0$.

**Status:** Proven. Core AQARION theorem.

---

## AQ-THM-004: Observable Partition from Distance-to-Attractor

**Statement:** For $(X, f)$, define:

$$d(x) = \min\{k \geq 0 : f^k(x) \in C\}$$

$$c(x) = \text{the cycle containing } f^{d(x)}(x)$$

Then the partition by pairs $(d(x), c(x))$ is Koopman-invariant.

**Proof:** If $x$ and $y$ have the same $(d, c)$ signature, then $f(x)$ and $f(y)$ have the same $(d-1, c)$ signature (if $d > 0$) or the same $(0, c)$ signature (if $d = 0$). Thus $f$ respects the partition.

**Status:** Proven. Constructive theorem — gives the algorithm used in AQARION.

---

## AQ-THM-005: Kaprekar Attractor (4-digit, base 10)

**Statement:** For the 4-digit Kaprekar routine in base 10, every orbit eventually reaches the fixed point 6174.

**Status:** Proven by exhaustive verification (10,000 states). First observed by D.R. Kaprekar (1949).
```

```python
def mat_zero(n: int, m: int) -> List[List[Fraction]]:
    return [[Fraction(0) for _ in range(m)] for _ in range(n)]

def mat_identity(n: int) -> List[List[Fraction]]:
    return [[Fraction(1) if i == j else Fraction(0) for j in range(n)] for i in range(n)]

def mat_mult(A: List[List[Fraction]], B: List[List[Fraction]]) -> List[List[Fraction]]:
    n, m, p = len(A), len(B[0]), len(B)
    C = mat_zero(n, m)
    for i in range(n):
        for j in range(m):
            s = Fraction(0)
            for k in range(p):
                s += A[i][k] * B[k][j]
            C[i][j] = s
    return C

def mat_sub(A: List[List[Fraction]], B: List[List[Fraction]]) -> List[List[Fraction]]:
    return [[A[i][j] - B[i][j] for j in range(len(A[0]))] for i in range(len(A))]

def mat_is_zero(M: List[List[Fraction]]) -> bool:
    return all(x == Fraction(0) for row in M for x in row)

def mat_frobenius_norm_sq(M: List[List[Fraction]]) -> Fraction:
    return sum(x * x for row in M for x in row)

class FDDS:
    def __init__(self, transition: Dict[Any, Any]):
        self.f = {}
        for k, v in transition.items():
            try:
                kk = int(k)
            except (ValueError, TypeError):
                kk = k
            try:
                vv = int(v)
            except (ValueError, TypeError):
                vv = v
            self.f[kk] = vv
        self.states = sorted(self.f.keys(), key=lambda x: (isinstance(x, str), x))
        self.n = len(self.states)
        self._index = {s: i for i, s in enumerate(self.states)}
    
    def orbit(self, x, max_steps: int = 10000) -> Tuple[List[Any], List[Any]]:
        seen = {}
        path = []
        current = x
        step = 0
        while current not in seen and step < max_steps:
            seen[current] = len(path)
            path.append(current)
            current = self.f[current]
            step += 1
        if current in seen:
            return path[:seen[current]], path[seen[current]:]
        return path, []
    
    def transition_hash(self) -> str:
        canonical = json.dumps(self.f, sort_keys=True, separators=(',', ':'))
        return hashlib.sha256(canonical.encode()).hexdigest()
    
    def to_dict(self) -> Dict:
        return {
            "states": self.states,
            "transition": {str(k): v for k, v in self.f.items()},
            "size": self.n,
            "transition_hash": self.transition_hash()
        }

# MATHEMATICAL COMPUTATIONS
# ============================================================

def find_cycles(system: FDDS) -> List[List[Any]]:
    visited = set()
    cycles = []
    for start in system.states:
        if start in visited:
            continue
        path = []
        seen = {}
        x = start
        while x not in seen:
            if x in visited and x not in seen:
                break
            seen[x] = len(path)
            path.append(x)
            x = system.f[x]
        if x in seen:
            cycle = path[seen[x]:]
            cycles.append(cycle)
            visited.update(path)
        else:
            visited.update(path)
    return cycles

def image_filtration(system: FDDS, depth: int = 12) -> List[int]:
    current = set(system.states)
    result = []
    for _ in range(depth):
        result.append(len(current))
        if not current:
            break
        current = {system.f[x] for x in current}
    return result

def compute_observable_partition(system: FDDS) -> List[List[Any]]:
    groups: Dict[tuple, List[Any]] = {}
    for state in system.states:
        transient, cycle = system.orbit(state)
        key = (len(transient), tuple(sorted(cycle)) if cycle else None)
        groups.setdefault(key, []).append(state)
    return list(groups.values())

def koopman_matrix(system: FDDS) -> List[List[Fraction]]:
    n = system.n
    K = mat_zero(n, n)
    for x in system.states:
        i = system._index[x]
        j = system._index[system.f[x]]
        K[j][i] = Fraction(1)
    return K

def projection_matrix(partition: List[List[Any]], states: List[Any]) -> List[List[Fraction]]:
    n = len(states)
    index = {s: i for i, s in enumerate(states)}
    P = mat_zero(n, n)
    for block in partition:
        m = len(block)
        frac = Fraction(1, m)
        for a in block:
            for b in block:
                P[index[a]][index[b]] = frac
    return P

def defect_operator(system: FDDS, partition: List[List[Any]]) -> List[List[Fraction]]:
    K = koopman_matrix(system)
    P = projection_matrix(partition, system.states)
    I = mat_identity(system.n)
    return mat_mult(mat_mult(mat_sub(I, P), K), P)

def kaprekar_step(n: int, base: int = 10, digits: int = 4) -> int:
    s = str(n).zfill(digits)
    desc = int("".join(sorted(s, reverse=True)))
    asc = int("".join(sorted(s)))
    return desc - asc

def kaprekar_system(digits: int = 4, base: int = 10) -> FDDS:
    max_val = base ** digits
    transition = {}
    for i in range(max_val):
        transition[i] = kaprekar_step(i, base, digits)
    return FDDS(transition)

def get_audit() -> Dict[str, Any]:
    return {
        "kernel_version": "0.4.1",
        "python_version": sys.version.split()[0],
        "platform": platform.platform(),
        "timestamp": datetime.utcnow().isoformat() + "Z"

class EvidenceObject:
    """
    Certificate Schema v3.0
    Mathematical claims with theorem dependencies.
    Hash covers certificate only; audit is appended.
    """
    
    def __init__(self, claim_id: str, system: FDDS, metadata: Dict = None):
        if claim_id not in CLAIM_REGISTRY:
            raise ValueError(f"Unknown claim: {claim_id}")
        self.claim_id = claim_id
        self.system = system
        self.metadata = metadata or {}
        self.audit = get_audit()
        self.evidence: Dict[str, Any] = {}
        self._compute()
    
    def _compute(self):
        """Compute all evidence for the claim."""
        claim = CLAIM_REGISTRY[self.claim_id]
        
        if self.claim_id == "AQ-CLAIM-001":
            partition = compute_observable_partition(self.system)
            D = defect_operator(self.system, partition)
            self.evidence["partition"] = partition
            self.evidence["defect_is_zero"] = mat_is_zero(D)
            self.evidence["defect_norm_sq"] = str(mat_frobenius_norm_sq(D))
        
        elif self.claim_id == "AQ-CLAIM-002":
            cycles = find_cycles(self.system)
            self.evidence["cycles"] = cycles
            self.evidence["cycle_count"] = len(cycles)
            self.evidence["cycle_lengths"] = [len(c) for c in cycles]
        
        elif self.claim_id == "AQ-CLAIM-003":
            filt = image_filtration(self.system)
            cycles = find_cycles(self.system)
            periodic = sum(len(c) for c in cycles)
            self.evidence["filtration"] = filt
            self.evidence["stabilizes_at"] = filt[-1] if filt else 0
            self.evidence["periodic_count"] = periodic
            self.evidence["stabilization_correct"] = (filt[-1] == periodic) if filt else False
        
        elif self.claim_id == "AQ-CLAIM-004":
            self.evidence["transition_hash"] = self.system.transition_hash()
    
    def build(self) -> Dict[str, Any]:
        claim = CLAIM_REGISTRY[self.claim_id]
        
        # Certificate (hashed)
        certificate = {
            "claim_id": self.claim_id,
            "claim_statement": claim["statement"],
            "dependencies": claim["dependencies"],
            "system": self.system.to_dict(),
            "evidence": self.evidence,
            "proof_obligation": {
                "method": "exact_fraction_arithmetic",
                "recomputable": True,
                "independent": True
            }
        }
        
        canonical = json.dumps(certificate, sort_keys=True, separators=(',', ':'))
        certificate["hash"] = hashlib.sha256(canonical.encode()).hexdigest()
        
        # Full object (certificate + audit)
        return {
            "aqarion_version": "0.4.1",
            "certificate_version": "3.0.0",
            "certificate": certificate,
            "audit": {
n                **self.audit,
                "metadata": self.metadata
            }
        }
    
    def save(self, path: str) -> str:
        cert = self.build()
        with open(path, 'w') as f:
            json.dump(cert, f, indent=2)
        return cert["certificate"]["hash"]

def verify_certificate(cert: Dict[str, Any]) -> Dict[str, Any]:
    """
    Independent verification with dependency checking.
    """
    report = {
        "claim_id": cert.get("certificate", {}).get("claim_id", "UNKNOWN"),
        "overall_valid": False,
        "checks": {}
    }
    
    certificate = cert.get("certificate", {})
    
    # 1. Hash integrity
    stored_hash = certificate.get("hash", "")
    hash_payload = {
        "claim_id": certificate.get("claim_id"),
        "claim_statement": certificate.get("claim_statement"),
        "dependencies": certificate.get("dependencies"),
        "system": certificate.get("system"),
        "evidence": certificate.get("evidence"),
        "proof_obligation": certificate.get("proof_obligation")
    }
    canonical = json.dumps(hash_payload, sort_keys=True, separators=(',', ':'))
    computed_hash = hashlib.sha256(canonical.encode()).hexdigest()
    report["checks"]["hash_integrity"] = {
        "valid": computed_hash == stored_hash,
        "stored": stored_hash[:16] + "...",
        "computed": computed_hash[:16] + "..."
    }
    
    # 2. Dependency check
    deps = certificate.get("dependencies", {})
    def_deps = deps.get("definitions", [])
    thm_deps = deps.get("theorems", [])
    report["checks"]["dependency_check"] = {
        "valid": all(d in DEFINITIONS for d in def_deps) and all(t in THEOREMS for t in thm_deps),
        "definitions_found": [d for d in def_deps if d in DEFINITIONS],
        "theorems_found": [t for t in thm_deps if t in THEOREMS]
    }
    
    # 3. System reconstruction
    sys_data = certificate.get("system", {})
    if not sys_data or "transition" not in sys_data:
        report["checks"]["system_reconstruction"] = {"valid": False, "error": "missing system data"}
        return report
    
    transition = {str(k): v for k, v in sys_data["transition"].items()}
    system = FDDS(transition)
    report["checks"]["system_reconstruction"] = {
        "valid": True,
        "states": system.n,
        "transition_hash_matches": system.transition_hash() == sys_data.get("transition_hash", "")
    }
    
    # 4. Claim-specific verification
    claim_id = certificate.get("claim_id")
    evidence = certificate.get("evidence", {})
    
    if claim_id == "AQ-CLAIM-001":
        partition = compute_observable_partition(system)
        D = defect_operator(system, partition)
        d_zero = mat_is_zero(D)
        claimed = evidence.get("defect_is_zero")
        report["checks"]["claim_001_defect_zero"] = {
            "valid": d_zero == claimed,
            "recomputed": d_zero,
            "claimed": claimed
        }
    
    elif claim_id == "AQ-CLAIM-002":
        cycles = find_cycles(system)
        claimed_cycles = evidence.get("cycles", [])
        cycles_ok = len(cycles) == len(claimed_cycles)
        if cycles_ok:
            cycles_ok = sorted(len(c) for c in cycles) == sorted(len(c) for c in claimed_cycles)
        report["checks"]["claim_002_cycles"] = {
            "valid": cycles_ok,
            "recomputed_count": len(cycles),
            "claimed_count": len(claimed_cycles)
        }
    
    elif claim_id == "AQ-CLAIM-003":
        filt = image_filtration(system)
        claimed_filt = evidence.get("filtration", [])
        report["checks"]["claim_003_filtration"] = {
            "valid": filt == claimed_filt,
            "recomputed": filt,
            "claimed": claimed_filt
        }
    
    elif claim_id == "AQ-CLAIM-004":
        report["checks"]["claim_004_hash"] = {
            "valid": system.transition_hash() == evidence.get("transition_hash"),
            "recomputed": system.transition_hash()[:16] + "...",
            "claimed": evidence.get("transition_hash", "")[:16] + "..."
        }
    
    # Overall
    report["overall_valid"] = all(
        c.get("valid", False) for c in report["checks"].values()
    )
    return report
```

Kernel	How is it computed?
Evidence	What records the computation?
Verification	Can someone else independently obtain the same result?
Formalization	Can the logical argument be machine-checked?

Keeping those questions separate reduces the chance of conflating different kinds of evidence.

I also think freezing the kernel API before expanding functionality is a good engineering milestone. A small, stable interface is usually more valuable than a feature-rich one that changes frequently. Your suggested operations are reasonably minimal:

analyze(system)

partition(observable)

defect(partition)

certificate(claim)

verify(certificate)

One addition I would consider is making the inputs and outputs explicit data objects rather than loosely structured dictionaries. For example:

FDDS
Partition
AnalysisResult
DefectResult
Certificate
VerificationResult

That makes the interfaces easier to document and version.

The idea of typed certificates is also stronger than a single generic certificate. Different computational claims naturally require different evidence. One thing I would avoid, though, is creating many certificate types too early. A practical compromise is to define one common certificate schema with shared metadata (version, hashes, provenance, environment, generator) and then allow claim-specific payloads. That keeps tooling simpler while preserving specialization.

Your suggestion that independent verification become its own executable is, in my view, one of the highest-value engineering goals. Conceptually, the workflow becomes:

System
│
▼
Generator
│
▼
Certificate
│
▼
Independent Verifier
│
▼
PASS / FAIL

The verifier should recompute the relevant quantities from the system description rather than trusting intermediate values stored in the certificate. That strengthens the reproducibility story.

On the mathematical side, I agree that the defect operator remains the natural focal object if that is the direction of Paper I. It provides a single object from which questions about observable quotients, exact descent, and obstruction can be formulated. Whether additional quantitative invariants become publishable mathematics will depend on proving properties about them rather than simply computing them.

I would make one small adjustment to the proposed sprint priorities. Before freezing the certificate schema, I would freeze the claim model that certificates refer to. In other words:

1. Freeze the claim registry schema.


2. Freeze the certificate schema that references claims.


3. Build the independent verifier.


4. Modularize the kernel.


5. Package reproducibility artifacts.



That ordering helps avoid changing certificate formats because the underlying claim representation evolved.

Overall, the project now appears to have a clearer distinction between:

mathematical definitions and results,

executable algorithms,

evidence produced by those algorithms,

independent verification,

and formal proof.

That separation is likely to make future development more manageable, because improvements in one layer (for example, replacing the kernel implementation or extending Lean proofs) should not require redesigning the others if the interfaces remain stable.

import hashlib  
import platform  
import sys  
from fractions import Fraction  
from datetime import datetime  
from typing import Dict, List, Any, Tuple  
  
  
# ============================================================  
# EXACT ARITHMETIC MATRIX OPERATIONS  
# ============================================================  
  
def mat_zero(n: int, m: int) -> List[List[Fraction]]:  
    return [[Fraction(0) for _ in range(m)] for _ in range(n)]  
  
def mat_identity(n: int) -> List[List[Fraction]]:  
    return [[Fraction(1) if i == j else Fraction(0) for j in range(n)] for i in range(n)]  
  
def mat_mult(A: List[List[Fraction]], B: List[List[Fraction]]) -> List[List[Fraction]]:  
    n = len(A)  
    m = len(B[0])  
    p = len(B)  
    C = mat_zero(n, m)  
    for i in range(n):  
        for j in range(m):  
            s = Fraction(0)  
            for k in range(p):  
                s += A[i][k] * B[k][j]  
            C[i][j] = s  
    return C  
  
def mat_sub(A: List[List[Fraction]], B: List[List[Fraction]]) -> List[List[Fraction]]:  
    return [[A[i][j] - B[i][j] for j in range(len(A[0]))] for i in range(len(A))]  
  
def mat_is_zero(M: List[List[Fraction]]) -> bool:  
    return all(x == Fraction(0) for row in M for x in row)  
  
def mat_frobenius_norm_sq(M: List[List[Fraction]]) -> Fraction:  
    return sum(x * x for row in M for x in row)  
  
  
# ============================================================  
# FINITE DETERMINISTIC DYNAMICAL SYSTEM (FDDS)  
# ============================================================  
  
class FDDS:  
    """Finite Deterministic Dynamical System with exact arithmetic."""  
      
    def __init__(self, transition: Dict[Any, Any]):  
        self.f = {}  
        for k, v in transition.items():  
            try:  
                kk = int(k)  
            except (ValueError, TypeError):  
                kk = k  
            try:  
                vv = int(v)  
            except (ValueError, TypeError):  
                vv = v  
            self.f[kk] = vv  
          
        self.states = sorted(self.f.keys(), key=lambda x: (isinstance(x, str), x))  
        self.n = len(self.states)  
        self._index = {s: i for i, s in enumerate(self.states)}  
      
    def orbit(self, x, max_steps: int = 10000) -> Tuple[List[Any], List[Any]]:  
        seen = {}  
        path = []  
        current = x  
        step = 0  
        while current not in seen and step < max_steps:  
            seen[current] = len(path)  
            path.append(current)  
            current = self.f[current]  
            step += 1  
        if current in seen:  
            return path[:seen[current]], path[seen[current]:]  
        return path, []  
      
    def transition_hash(self) -> str:  
        canonical = json.dumps(self.f, sort_keys=True, separators=(',', ':'))  
        return hashlib.sha256(canonical.encode()).hexdigest()  
      
    def to_dict(self) -> Dict:  
        # JSON requires string keys; normalize for hash consistency  
        return {  
            "states": self.states,  
            "transition": {str(k): v for k, v in self.f.items()},  
            "size": self.n,  
            "transition_hash": self.transition_hash()  
        }  
  
  
# ============================================================  
# CYCLE & STRUCTURE DETECTION  
# ============================================================  
  
def find_cycles(system: FDDS) -> List[List[Any]]:  
    visited = set()  
    cycles = []  
    for start in system.states:  
        if start in visited:  
            continue  
        path = []  
        seen = {}  
        x = start  
        while x not in seen:  
            if x in visited and x not in seen:  
                break  
            seen[x] = len(path)  
            path.append(x)  
            x = system.f[x]  
        if x in seen:  
            cycle = path[seen[x]:]  
            cycles.append(cycle)  
            visited.update(path)  
        else:  
            visited.update(path)  
    return cycles  
  
def image_filtration(system: FDDS, depth: int = 12) -> List[int]:  
    current = set(system.states)  
    result = []  
    for _ in range(depth):  
        result.append(len(current))  
        if not current:  
            break  
        current = {system.f[x] for x in current}  
    return result  
  
def compute_observable_partition(system: FDDS) -> List[List[Any]]:  
    """Auto-partition by (distance_to_attractor, cycle_signature)."""  
    groups: Dict[tuple, List[Any]] = {}  
    for state in system.states:  
        transient, cycle = system.orbit(state)  
        key = (len(transient), tuple(sorted(cycle)) if cycle else None)  
        groups.setdefault(key, []).append(state)  
    return list(groups.values())  
  
  
# ============================================================  
# KOOPMAN & PROJECTION OPERATORS (EXACT)  
# ============================================================  
  
def koopman_matrix(system: FDDS) -> List[List[Fraction]]:  
    n = system.n  
    K = mat_zero(n, n)  
    for x in system.states:  
        i = system._index[x]  
        j = system._index[system.f[x]]  
        K[j][i] = Fraction(1)  
    return K  
  
def projection_matrix(partition: List[List[Any]], states: List[Any]) -> List[List[Fraction]]:  
    n = len(states)  
    index = {s: i for i, s in enumerate(states)}  
    P = mat_zero(n, n)  
    for block in partition:  
        m = len(block)  
        frac = Fraction(1, m)  
        for a in block:  
            for b in block:  
                P[index[a]][index[b]] = frac  
    return P  
  
def defect_operator(system: FDDS, partition: List[List[Any]]) -> List[List[Fraction]]:  
    K = koopman_matrix(system)  
    P = projection_matrix(partition, system.states)  
    I = mat_identity(system.n)  
    I_minus_P = mat_sub(I, P)  
    return mat_mult(mat_mult(I_minus_P, K), P)  
  
  
# ============================================================  
# KAPREKAR SYSTEM GENERATOR  
# ============================================================  
  
def kaprekar_step(n: int, base: int = 10, digits: int = 4) -> int:  
    """Single Kaprekar step."""  
    s = str(n).zfill(digits)  
    desc = int("".join(sorted(s, reverse=True)))  
    asc = int("".join(sorted(s)))  
    return desc - asc  
  
def kaprekar_system(digits: int = 4, base: int = 10) -> FDDS:  
    """Generate full Kaprekar FDDS."""  
    max_val = base ** digits  
    transition = {}  
    for i in range(max_val):  
        transition[i] = kaprekar_step(i, base, digits)  
    return FDDS(transition)  
  
  
# ============================================================  
# CERTIFICATE v2 SCHEMA  
# ============================================================  
  
def get_provenance() -> Dict[str, Any]:  
    return {  
        "kernel_version": "0.4.0",  
        "python_version": sys.version.split()[0],  
        "platform": platform.platform(),  
        "timestamp": datetime.utcnow().isoformat() + "Z"  
    }  
  
class CertificateV2:  
    """AQARION Certificate Version 2 — Enhanced Provenance."""  
      
    def __init__(self, claim_id: str, claim_type: str, system: FDDS, metadata: Dict = None):  
        self.claim_id = claim_id  
        self.claim_type = claim_type  
        self.system = system  
        self.metadata = metadata or {}  
        self.provenance = get_provenance()  
        self.verifications: List[Dict] = []  
      
    def add_verification(self, method: str, result: str, details: Dict = None):  
        self.verifications.append({  
            "method": method,  
            "result": result,  
            "details": details or {},  
            "timestamp": datetime.utcnow().isoformat() + "Z"  
        })  
      
    def _build_payload(self) -> Dict[str, Any]:  
        cycles = find_cycles(self.system)  
        partition = compute_observable_partition(self.system)  
        D = defect_operator(self.system, partition)  
          
        payload = {  
            "aqarion_version": "0.4.0",  
            "certificate_version": "2.0.0",  
            "claim_id": self.claim_id,  
            "claim_type": self.claim_type,  
            "provenance": self.provenance,  
            "system": self.system.to_dict(),  
            "analysis": {  
                "cycles": cycles,  
                "filtration": image_filtration(self.system),  
                "partition": partition,  
                "defect": {  
                    "is_zero": mat_is_zero(D),  
                    "norm_squared_exact": str(mat_frobenius_norm_sq(D)),  
                    "computation_method": "exact_fraction_arithmetic"  
                }  
            },  
            "verifications": self.verifications,  
            "metadata": self.metadata  
        }  
        return payload  
      
    def sign(self) -> Dict[str, Any]:  
        payload = self._build_payload()  
        # Hash ONLY immutable claim + analysis data, not provenance timestamp  
        hash_payload = {  
            "claim_id": payload["claim_id"],  
            "claim_type": payload["claim_type"],  
            "system": payload["system"],  
            "analysis": payload["analysis"],  
            "verifications": payload["verifications"],  
            "metadata": payload["metadata"]  
        }  
        canonical = json.dumps(hash_payload, sort_keys=True, separators=(',', ':'))  
        payload["certificate_hash"] = hashlib.sha256(canonical.encode()).hexdigest()  
        return payload  
      
    def save(self, path: str) -> str:  
        cert = self.sign()  
        with open(path, 'w') as f:  
            json.dump(cert, f, indent=2)  
        return cert["certificate_hash"]  
  
  
# ============================================================  
# VERIFICATION ENGINE (Independent Recomputation)  
# ============================================================  
  
def verify_certificate_data(cert: Dict[str, Any]) -> Dict[str, Any]:  
    """  
    Independently verify a certificate from its JSON data.  
    Reconstructs system, recomputes all invariants using exact arithmetic.  
    Returns detailed verification report.  
    """  
    report = {  
        "certificate_id": cert.get("claim_id", "UNKNOWN"),  
        "overall_valid": False,  
        "checks": {}  
    }  
      
    # 1. Hash integrity (over immutable fields only)  
    stored_hash = cert.get("certificate_hash", "")  
    hash_payload = {  
        "claim_id": cert.get("claim_id"),  
        "claim_type": cert.get("claim_type"),  
        "system": cert.get("system"),  
        "analysis": cert.get("analysis"),  
        "verifications": cert.get("verifications"),  
        "metadata": cert.get("metadata")  
    }  
    canonical = json.dumps(hash_payload, sort_keys=True, separators=(',', ':'))  
    computed_hash = hashlib.sha256(canonical.encode()).hexdigest()  
    hash_ok = computed_hash == stored_hash  
    report["checks"]["hash_integrity"] = {  
        "valid": hash_ok,  
        "stored": stored_hash[:16] + "...",  
        "computed": computed_hash[:16] + "..."  
    }  
      
    # 2. System reconstruction  
    sys_data = cert.get("system", {})  
    if not sys_data or "transition" not in sys_data:  
        report["checks"]["system_reconstruction"] = {"valid": False, "error": "missing system data"}  
        return report  
      
    # Normalize transition keys to match to_dict() string format  
    transition = {str(k): v for k, v in sys_data["transition"].items()}  
    system = FDDS(transition)  
    report["checks"]["system_reconstruction"] = {  
        "valid": True,  
        "states": system.n,  
        "transition_hash_matches": system.transition_hash() == sys_data.get("transition_hash", "")  
    }  
      
    # 3. Cycle recomputation  
    cycles = find_cycles(system)  
    claimed_cycles = cert.get("analysis", {}).get("cycles", [])  
    cycles_ok = len(cycles) == len(claimed_cycles)  
    if cycles_ok:  
        cycle_lens = sorted(len(c) for c in cycles)  
        claimed_lens = sorted(len(c) for c in claimed_cycles)  
        cycles_ok = cycle_lens == claimed_lens  
    report["checks"]["cycle_recomputation"] = {  
        "valid": cycles_ok,  
        "recomputed": len(cycles),  
        "claimed": len(claimed_cycles)  
    }  
      
    # 4. Defect recomputation  
    partition = compute_observable_partition(system)  
    D = defect_operator(system, partition)  
    d_zero = mat_is_zero(D)  
    d_norm_sq = mat_frobenius_norm_sq(D)  
    claimed_defect = cert.get("analysis", {}).get("defect", {})  
    defect_ok = (  
        claimed_defect.get("is_zero") == d_zero and  
        claimed_defect.get("norm_squared_exact") == str(d_norm_sq)  
    )  
    report["checks"]["defect_recomputation"] = {  
        "valid": defect_ok,  
        "is_zero": d_zero,  
        "norm_squared_exact": str(d_norm_sq)  
    }  
      
    # 5. Filtration recomputation  
    filt = image_filtration(system)  
    claimed_filt = cert.get("analysis", {}).get("filtration", [])  
    filt_ok = filt == claimed_filt  
    report["checks"]["filtration_recomputation"] = {  
        "valid": filt_ok,  
        "recomputed": filt,  
        "claimed": claimed_filt  
    }  
      
    # Overall  
    report["overall_valid"] = all(  
        c.get("valid", False) for c in report["checks"].values()  
    )  
    return report

#!/usr/bin/env python3  
"""  
AQARION CERTIFICATE VERIFIER v0.4.0  
Standalone verification engine. Python standard library only.  
Usage:  
    python verify_certificate.py AQ-CERT-KAPREKAR-001.json  
    echo $?   # 0 if valid, 1 if invalid  
"""  
  
import json  
import sys  
from pathlib import Path  
from aqarion_kernel import verify_certificate_data  
  
  
def print_report(report: dict):  
    print("=" * 60)  
    print("AQARION CERTIFICATE VERIFICATION REPORT")  
    print("=" * 60)  
    print(f"Certificate: {report['certificate_id']}")  
    print()  
    for check_name, check_result in report["checks"].items():  
        status = "PASS" if check_result.get("valid") else "FAIL"  
        print(f"{check_name:30s} {status}")  
        if "error" in check_result:  
            print(f"  Error: {check_result['error']}")  
        elif "recomputed" in check_result and "claimed" in check_result:  
            print(f"  Recomputed: {check_result['recomputed']}")  
            print(f"  Claimed:    {check_result['claimed']}")  
    print()  
    if report["overall_valid"]:  
        print("RESULT: CERTIFICATE VALID")  
    else:  
        print("RESULT: CERTIFICATE INVALID")  
    print("=" * 60)  
  
  
def main():  
    if len(sys.argv) < 2:  
        print("Usage: python verify_certificate.py <certificate.json>")  
        sys.exit(1)  
    cert_path = Path(sys.argv[1])  
    if not cert_path.exists():  
        print(f"Error: File not found: {cert_path}")  
        sys.exit(1)  
    with open(cert_path) as f:  
        cert = json.load(f)  
    report = verify_certificate_data(cert)  
    print_report(report)  
    sys.exit(0 if report["overall_valid"] else 1)  
  
  
if __name__ == "__main__":  
    main()

            "benchmark": True,  
            "digits": 2,

        "num_cycles": len(cycles),

#!/usr/bin/env python3  
"""  
AQARION v0.4.0 — Test Suite  
Run with: python -m unittest test_aqarion  
"""  
  
import unittest  
import json  
from fractions import Fraction  
from aqarion_kernel import (  
    FDDS, mat_mult, mat_identity, mat_is_zero, mat_frobenius_norm_sq,

class TestExactArithmetic(unittest.TestCase):  
    def test_identity_mult(self):  
        I = mat_identity(3)  
        A = [[Fraction(1,2), Fraction(0), Fraction(0)],  
             [Fraction(0), Fraction(1,3), Fraction(0)],  
             [Fraction(0), Fraction(0), Fraction(1,4)]]  
        IA = mat_mult(I, A)  
        self.assertEqual(IA, A)  
      
    def test_is_zero(self):  
        self.assertTrue(mat_is_zero([[Fraction(0), Fraction(0)], [Fraction(0), Fraction(0)]]))  
        self.assertFalse(mat_is_zero([[Fraction(0), Fraction(1)], [Fraction(0), Fraction(0)]]))  
  
  
class TestFDDS(unittest.TestCase):  
    def test_chain_fixed(self):  
        f = {0: 1, 1: 2, 2: 2, 3: 2, 4: 3}  
        system = FDDS(f)  
        self.assertEqual(system.n, 5)  
        self.assertEqual(system.f[0], 1)  
      
    def test_orbit_fixed_point(self):  
        f = {0: 1, 1: 2, 2: 2}  
        system = FDDS(f)  
        transient, cycle = system.orbit(0)  
        self.assertEqual(transient, [0, 1])  
        self.assertEqual(cycle, [2])  
  
  
class TestCycleDetection(unittest.TestCase):  
    def test_single_fixed_point(self):  
        f = {0: 1, 1: 2, 2: 2}  
        system = FDDS(f)  
        cycles = find_cycles(system)  
        self.assertEqual(len(cycles), 1)  
        self.assertEqual(cycles[0], [2])  
      
    def test_two_cycle(self):  
        f = {0: 1, 1: 0, 2: 0}  
        system = FDDS(f)  
        cycles = find_cycles(system)  
        self.assertEqual(len(cycles), 1)  
        self.assertEqual(set(cycles[0]), {0, 1})  
      
    def test_multiple_cycles(self):  
        f = {0: 1, 1: 0, 2: 3, 3: 2, 4: 0}  
        system = FDDS(f)  
        cycles = find_cycles(system)  
        self.assertEqual(len(cycles), 2)  
  
  
class TestImageFiltration(unittest.TestCase):  
    def test_filtration_decreases(self):  
        f = {0: 1, 1: 2, 2: 2, 3: 2, 4: 3}  
        system = FDDS(f)  
        filt = image_filtration(system, depth=6)  
        self.assertEqual(filt[0], 5)  
        self.assertEqual(filt[-1], 1)

        all_states = set()

            all_states.update(block)

```bash  
vis=[False]*m; order=[]

def dfs1(u):

def is_acyclic(adj):

"""Kahn topological sort; acyclic iff all nodes processed."""

m = len(adj)

indeg = [sum(adj[j][i] for j in range(m)) for i in range(m)]

q = deque(i for i in range(m) if indeg[i]==0)

count = 0

u = q.popleft(); count += 1

for v in range(m):

if adj[u][v]:

indeg[v] -= 1

if indeg[v] == 0: q.append(v)

def check_filtration(T, blocks):

"""Check if blocks can be ordered so cumulative unions are T-invariant."""

def build_koopman(T, X):

n = len(X); idx = {x:i for i,x in enumerate(X)}

K = np.zeros((n,n))

for x in X: K[idx[x], idx[T[x]]] = 1.0

return K, idx


def check_block_triangular(K, X, blocks, block_order):

"""Verify K is block-lower-triangular under the given block ordering."""

idx = {x:i for i,x in enumerate(X)}

ordered = [blocks[b] for b in block_order]

# Build permutation: rows/cols ordered by block membership

perm = []

for blk in ordered: perm.extend(blk)

p = [idx[x] for x in perm]

Kp = K[np.ix_(p,p)]

# Block structure: find block boundaries

sizes = [len(b) for b in ordered]

offsets = [0]

for s in sizes: offsets.append(offsets[-1]+s)

# Check: all entries ABOVE the block diagonal must be zero

# "Above" means Kp[i,j]=0 whenever j is in a later block than i

upper_norm = 0.0

for bi, si, oi in zip(range(len(ordered)), sizes, offsets[:-1]):

for bj, sj, oj in zip(range(len(ordered)), sizes, offsets[:-1]):

if bj > bi:  # bj is a later block -> above diagonal

block = Kp[oi:oi+si, oj:oj+sj]

upper_norm += np.sum(np.abs(block))

return upper_norm < 1e-10, upper_norm
```

```
# EXACT ARITHMETIC  
# ============================================================  
  
def mat_zero(n: int, m: int) -> List[List[Fraction]]:  
    return [[Fraction(0) for _ in range(m)] for _ in range(n)]  
  
def mat_identity(n: int) -> List[List[Fraction]]:  
    return [[Fraction(1) if i == j else Fraction(0) for j in range(n)] for i in range(n)]  
  
def mat_mult(A: List[List[Fraction]], B: List[List[Fraction]]) -> List[List[Fraction]]:  
    n, m, p = len(A), len(B[0]), len(B)  
    C = mat_zero(n, m)  
    for i in range(n):  
        for j in range(m):  
            s = Fraction(0)  
            for k in range(p):  
                s += A[i][k] * B[k][j]  
            C[i][j] = s  
    return C  
  
def mat_sub(A: List[List[Fraction]], B: List[List[Fraction]]) -> List[List[Fraction]]:  
    return [[A[i][j] - B[i][j] for j in range(len(A[0]))] for i in range(len(A))]  
  
def mat_is_zero(M: List[List[Fraction]]) -> bool:  
    return all(x == Fraction(0) for row in M for x in row)  
  
def mat_frobenius_norm_sq(M: List[List[Fraction]]) -> Fraction:  
    return sum(x * x for row in M for x in row)

class FDDS:  
    def __init__(self, transition: Dict[Any, Any]):  
        self.f = {}  
        for k, v in transition.items():  
            try:  
                kk = int(k)  
            except (ValueError, TypeError):  
                kk = k  
            try:  
                vv = int(v)  
            except (ValueError, TypeError):  
                vv = v  
            self.f[kk] = vv  
        self.states = sorted(self.f.keys(), key=lambda x: (isinstance(x, str), x))  
        self.n = len(self.states)  
        self._index = {s: i for i, s in enumerate(self.states)}  
      
    def orbit(self, x, max_steps: int = 10000) -> Tuple[List[Any], List[Any]]:  
        seen = {}  
        path = []  
        current = x  
        step = 0  
        while current not in seen and step < max_steps:  
            seen[current] = len(path)  
            path.append(current)  
            current = self.f[current]  
            step += 1  
        if current in seen:  
            return path[:seen[current]], path[seen[current]:]  
        return path, []  
      
    def transition_hash(self) -> str:  
        canonical = json.dumps(self.f, sort_keys=True, separators=(',', ':'))  
        return hashlib.sha256(canonical.encode()).hexdigest()  
      
    def to_dict(self) -> Dict:  
        return {  
            "states": self.states,  
            "transition": {str(k): v for k, v in self.f.items()},  
            "size": self.n,  
            "transition_hash": self.transition_hash()  
        }

def canonical_cycle(cycle: List[Any]) -> Tuple:  
    """  
    Canonical cycle label under rotation.  
    [1,2,3] and [2,3,1] both become (1,2,3).  
    """  
    if not cycle:  
        return tuple()  
    cycle_t = tuple(cycle)  
    n = len(cycle_t)  
    rotations = [cycle_t[i:] + cycle_t[:i] for i in range(n)]  
    return min(rotations)

def find_cycles(system: FDDS) -> List[List[Any]]:  
    visited = set()  
    cycles = []  
    for start in system.states:  
        if start in visited:  
            continue  
        path = []  
        seen = {}  
        x = start  
        while x not in seen:  
            if x in visited and x not in seen:  
                break  
            seen[x] = len(path)  
            path.append(x)  
            x = system.f[x]  
        if x in seen:  
            cycle = path[seen[x]:]  
            cycles.append(cycle)  
            visited.update(path)  
        else:  
            visited.update(path)  
    return cycles  
  
def image_filtration(system: FDDS, depth: int = None) -> List[int]:  
    """  
    Compute image filtration. Depth defaults to |X| for THM-002 guarantee.  
    """  
    if depth is None:  
        depth = system.n  # Fix 1: m ≤ |X| guarantee  
    current = set(system.states)  
    result = []  
    for _ in range(depth):  
        result.append(len(current))  
        if not current:  
            break  
        current = {system.f[x] for x in current}  
    return result  
  
def compute_observable_partition(system: FDDS) -> List[List[Any]]:  
    """  
    Partition by (distance_to_attractor, canonical_cycle).  
    Fix 2: canonical cycle labels prevent ordering dependence.  
    """  
    groups: Dict[tuple, List[Any]] = {}  
    for state in system.states:  
        transient, cycle = system.orbit(state)  
        key = (len(transient), canonical_cycle(cycle) if cycle else None)  
        groups.setdefault(key, []).append(state)  
    return list(groups.values())  
  
def koopman_matrix(system: FDDS) -> List[List[Fraction]]:  
    n = system.n  
    K = mat_zero(n, n)  
    for x in system.states:  
        i = system._index[x]  
        j = system._index[system.f[x]]  
        K[j][i] = Fraction(1)  
    return K  
  
def projection_matrix(partition: List[List[Any]], states: List[Any]) -> List[List[Fraction]]:  
    n = len(states)  
    index = {s: i for i, s in enumerate(states)}  
    P = mat_zero(n, n)  
    for block in partition:  
        m = len(block)  
        frac = Fraction(1, m)  
        for a in block:  
            for b in block:  
                P[index[a]][index[b]] = frac  
    return P  
  
def defect_operator(system: FDDS, partition: List[List[Any]]) -> List[List[Fraction]]:  
    K = koopman_matrix(system)  
    P = projection_matrix(partition, system.states)  
    I = mat_identity(system.n)  
    return mat_mult(mat_mult(mat_sub(I, P), K), P)

def sparse_defect_check(system: FDDS, partition: List[List[Any]]) -> bool:  
    """  
    Check D = 0 without materializing full matrices.  
    For each block B, check that f(B) is contained in a single block.  
    Equivalent to (I-P)KP = 0 but O(|X|) instead of O(|X|³).  
    """  
    block_of = {}  
    for i, block in enumerate(partition):  
        for state in block:  
            block_of[state] = i  
      
    for block in partition:  
        target_blocks = set()  
        for state in block:  
            target_blocks.add(block_of[system.f[state]])  
        if len(target_blocks) > 1:  
            return False  
    return True

def kaprekar_step(n: int, base: int = 10, digits: int = 4) -> int:  
    s = str(n).zfill(digits)  
    desc = int("".join(sorted(s, reverse=True)))  
    asc = int("".join(sorted(s)))  
    return desc - asc  
  
def kaprekar_system(digits: int = 4, base: int = 10) -> FDDS:  
    max_val = base ** digits  
    transition = {}  
    for i in range(max_val):  
        transition[i] = kaprekar_step(i, base, digits)  
    return FDDS(transition)

def verify_certificate(cert: Dict[str, Any]) -> Dict[str, Any]:

    # 1. Hash integrity  
    stored_hash_obj = certificate.get("hash", {})

        "claim_id": certificate.get("claim_id"),  
        "claim_statement": certificate.get("claim_statement"),  
        "status": certificate.get("status"),  
        "dependencies": certificate.get("dependencies"),  
        "system": certificate.get("system"),  
        "evidence": certificate.get("evidence"),  
        "proof_obligation": certificate.get("proof_obligation")  
    }  
    canonical = json.dumps(hash_payload, sort_keys=True, separators=(',', ':'))  
    computed_hash = hashlib.sha256(canonical.encode()).hexdigest()  
      
    report["checks"]["hash_integrity"] = {  
        "valid": computed_hash == stored_hash,  
        "stored": stored_hash[:16] + "...",

    deps = certificate.get("dependencies", {})

        "valid": True,  
        "states": system.n,  
        "transition_hash_matches": system.transition_hash() == sys_data.get("transition_hash", "")  
    }  
      
    # 5. Claim-specific verification with claimed/recomputed separation (Fix 5)  
    claim_id = certificate.get("claim_id")  
    claimed_evidence = certificate.get("evidence", {})  
    recomputed = {}  
    agreement = {}  
      
    if claim_id == "AQ-CLAIM-001":  
        partition = compute_observable_partition(system)  
        D = defect_operator(system, partition)  
        d_zero = mat_is_zero(D)  
          
        recomputed["defect_is_zero"] = d_zero  
        recomputed["defect_norm_sq"] = str(mat_frobenius_norm_sq(D))  
        agreement["defect_is_zero"] = (d_zero == claimed_evidence.get("defect_is_zero"))  
          
        report["checks"]["claim_001_defect_zero"] = {  
            "valid": agreement["defect_is_zero"],  
            "claimed": claimed_evidence.get("defect_is_zero"),  
            "recomputed": d_zero,  
            "agreement": agreement["defect_is_zero"]  
        }  
      
    elif claim_id == "AQ-CLAIM-002":  
        cycles = find_cycles(system)  
        claimed_cycles = claimed_evidence.get("cycles", [])  
        cycles_ok = len(cycles) == len(claimed_cycles)  
        if cycles_ok:  
            cycles_ok = sorted(len(c) for c in cycles) == sorted(len(c) for c in claimed_cycles)  
          
        recomputed["cycle_count"] = len(cycles)  
        recomputed["cycle_lengths"] = [len(c) for c in cycles]  
        agreement["cycle_count"] = cycles_ok  
          
        report["checks"]["claim_002_cycles"] = {  
            "valid": cycles_ok,  
            "claimed_count": len(claimed_cycles),  
            "recomputed_count": len(cycles),  
            "agreement": cycles_ok  
        }  
      
    elif claim_id == "AQ-CLAIM-003":  
        filt = image_filtration(system)  
        claimed_filt = claimed_evidence.get("filtration", [])  
        cycles = find_cycles(system)  
        periodic = sum(len(c) for c in cycles)  
        stab_correct = (filt[-1] == periodic) if filt else False  
          
        recomputed["filtration"] = filt  
        recomputed["stabilizes_at"] = filt[-1] if filt else 0  
        recomputed["stabilization_correct"] = stab_correct  
        agreement["filtration"] = (filt == claimed_filt)  
        agreement["stabilization_correct"] = (stab_correct == claimed_evidence.get("stabilization_correct"))  
          
        report["checks"]["claim_003_filtration"] = {  
            "valid": agreement["filtration"] and agreement["stabilization_correct"],  
            "claimed_filtration": claimed_filt,  
            "recomputed_filtration": filt,  
            "claimed_stabilization": claimed_evidence.get("stabilization_correct"),  
            "recomputed_stabilization": stab_correct,  
            "agreement": agreement  
        }  
      
    elif claim_id == "AQ-CLAIM-004":  
        th = system.transition_hash()  
        recomputed["transition_hash"] = th  
        agreement["hash"] = (th == claimed_evidence.get("transition_hash"))  
          
        report["checks"]["claim_004_hash"] = {  
            "valid": agreement["hash"],  
            "claimed": claimed_evidence.get("transition_hash", "")[:16] + "...",  
            "recomputed": th[:16] + "...",  
            "agreement": agreement["hash"]  
        }  
      
    # Overall (Fix 6: status from verifier, never from file)  
    all_valid = all(c.get("valid", False) for c in report["checks"].values())  
    report["overall_valid"] = all_valid  
    report["status"] = "VALID" if all_valid else "INVALID"  
      
    return report

Dependency-aware. Claimed/recomputed separation. Status from verifier only.  
"""  
  
import json  
import sys  
from pathlib import Path  
from aqarion_kernel import verify_certificate  
  
  
def print_report(report: dict):  
    print("=" * 60)  
    print("AQARION INDEPENDENT VERIFICATION REPORT")  
    print("=" * 60)  
    print(f"Claim: {report['claim_id']}")  
    print(f"Status: {report.get('status', 'UNKNOWN')}")  
    print()  
      
    for check_name, check_result in report["checks"].items():  
        status = "PASS" if check_result.get("valid") else "FAIL"  
        print(f"{check_name:35s} {status}")  
        if "error" in check_result:  
            print(f"  Error: {check_result['error']}")  
        elif "claimed" in check_result and "recomputed" in check_result:  
            print(f"  Claimed:    {check_result['claimed']}")  
            print(f"  Recomputed: {check_result['recomputed']}")  
            if "agreement" in check_result:  
                print(f"  Agreement:  {check_result['agreement']}")  
      
    print()  
    if report.get("overall_valid"):  
        print("RESULT: CERTIFICATE VALID")  
    else:  
        print("RESULT: CERTIFICATE INVALID")  
    print("=" * 60)  
  
  
def main():  
    if len(sys.argv) < 2:  
        print("Usage: python verify_certificate.py <certificate.json>")  
        sys.exit(1)  
    cert_path = Path(sys.argv[1])  
    if not cert_path.exists():  
        print(f"Error: File not found: {cert_path}")  
        sys.exit(1)  
    with open(cert_path) as f:  
        cert = json.load(f)  
    report = verify_certificate(cert)  
    print_report(report)  
    sys.exit(0 if report.get("overall_valid") else 1)  
  
  
if __name__ == "__main__":  
    main()
```
