# ============================================================  
# EXACT ARITHMETIC  
# ============================================================  
  
def mat_zero(n: int, m: int) -> List[List[Fraction]]:  
    return [[Fraction(0) for _ in range(m)] for _ in range(n)]  
  
def mat_identity(n: int) -> List[List[Fraction]]:  
    return [[Fraction(1) if i == j else Fraction(0) for j in range(n)] for i in range(n)]  
  
def mat_mult(A: List[List[Fraction]], B: List[List[Fraction]]) -> List[List[Fraction]]:  
    n, m, p = len(A), len(B[0]), len(B)  
    C = mat_zero(n, m)  
    for i in range(n):  
        for j in range(m):  
            s = Fraction(0)  
            for k in range(p):  
                s += A[i][k] * B[k][j]  
            C[i][j] = s  
    return C  
  
def mat_sub(A: List[List[Fraction]], B: List[List[Fraction]]) -> List[List[Fraction]]:  
    return [[A[i][j] - B[i][j] for j in range(len(A[0]))] for i in range(len(A))]  
  
def mat_is_zero(M: List[List[Fraction]]) -> bool:  
    return all(x == Fraction(0) for row in M for x in row)  
  
def mat_frobenius_norm_sq(M: List[List[Fraction]]) -> Fraction:  
    return sum(x * x for row in M for x in row)

class FDDS:  
    def __init__(self, transition: Dict[Any, Any]):  
        self.f = {}  
        for k, v in transition.items():  
            try:  
                kk = int(k)  
            except (ValueError, TypeError):  
                kk = k  
            try:  
                vv = int(v)  
            except (ValueError, TypeError):  
                vv = v  
            self.f[kk] = vv  
        self.states = sorted(self.f.keys(), key=lambda x: (isinstance(x, str), x))  
        self.n = len(self.states)  
        self._index = {s: i for i, s in enumerate(self.states)}  
      
    def orbit(self, x, max_steps: int = 10000) -> Tuple[List[Any], List[Any]]:  
        seen = {}  
        path = []  
        current = x  
        step = 0  
        while current not in seen and step < max_steps:  
            seen[current] = len(path)  
            path.append(current)  
            current = self.f[current]  
            step += 1  
        if current in seen:  
            return path[:seen[current]], path[seen[current]:]  
        return path, []  
      
    def transition_hash(self) -> str:  
        canonical = json.dumps(self.f, sort_keys=True, separators=(',', ':'))  
        return hashlib.sha256(canonical.encode()).hexdigest()  
      
    def to_dict(self) -> Dict:  
        return {  
            "states": self.states,  
            "transition": {str(k): v for k, v in self.f.items()},  
            "size": self.n,  
            "transition_hash": self.transition_hash()  
        }

# CANONICAL CYCLE (Fix 2)

def canonical_cycle(cycle: List[Any]) -> Tuple:  
    """  
    Canonical cycle label under rotation.  
    [1,2,3] and [2,3,1] both become (1,2,3).  
    """  
    if not cycle:  
        return tuple()  
    cycle_t = tuple(cycle)  
    n = len(cycle_t)  
    rotations = [cycle_t[i:] + cycle_t[:i] for i in range(n)]  
    return min(rotations)

def find_cycles(system: FDDS) -> List[List[Any]]:  
    visited = set()  
    cycles = []  
    for start in system.states:  
        if start in visited:  
            continue  
        path = []  
        seen = {}  
        x = start  
        while x not in seen:  
            if x in visited and x not in seen:  
                break  
            seen[x] = len(path)  
            path.append(x)  
            x = system.f[x]  
        if x in seen:  
            cycle = path[seen[x]:]  
            cycles.append(cycle)  
            visited.update(path)  
        else:  
            visited.update(path)  
    return cycles  
  
def image_filtration(system: FDDS, depth: int = None) -> List[int]:  
    """  
    Compute image filtration. Depth defaults to |X| for THM-002 guarantee.  
    """  
    if depth is None:  
        depth = system.n  # Fix 1: m ≤ |X| guarantee  
    current = set(system.states)  
    result = []  
    for _ in range(depth):  
        result.append(len(current))  
        if not current:  
            break  
        current = {system.f[x] for x in current}  
    return result  
  
def compute_observable_partition(system: FDDS) -> List[List[Any]]:  
    """  
    Partition by (distance_to_attractor, canonical_cycle).  
    Fix 2: canonical cycle labels prevent ordering dependence.  
    """  
    groups: Dict[tuple, List[Any]] = {}  
    for state in system.states:  
        transient, cycle = system.orbit(state)  
        key = (len(transient), canonical_cycle(cycle) if cycle else None)  
        groups.setdefault(key, []).append(state)  
    return list(groups.values())  
  
def koopman_matrix(system: FDDS) -> List[List[Fraction]]:  
    n = system.n  
    K = mat_zero(n, n)  
    for x in system.states:  
        i = system._index[x]  
        j = system._index[system.f[x]]  
        K[j][i] = Fraction(1)  
    return K  
  
def projection_matrix(partition: List[List[Any]], states: List[Any]) -> List[List[Fraction]]:  
    n = len(states)  
    index = {s: i for i, s in enumerate(states)}  
    P = mat_zero(n, n)  
    for block in partition:  
        m = len(block)  
        frac = Fraction(1, m)  
        for a in block:  
            for b in block:  
                P[index[a]][index[b]] = frac  
    return P  
  
def defect_operator(system: FDDS, partition: List[List[Any]]) -> List[List[Fraction]]:  
    K = koopman_matrix(system)  
    P = projection_matrix(partition, system.states)  
    I = mat_identity(system.n)  
    return mat_mult(mat_mult(mat_sub(I, P), K), P)

# SPARSE DEFECT CHECK (v0.5.0 Ready)  
# ============================================================  
  
def sparse_defect_check(system: FDDS, partition: List[List[Any]]) -> bool:  
    """  
    Check D = 0 without materializing full matrices.  
    For each block B, check that f(B) is contained in a single block.  
    Equivalent to (I-P)KP = 0 but O(|X|) instead of O(|X|³).  
    """  
    block_of = {}  
    for i, block in enumerate(partition):  
        for state in block:  
            block_of[state] = i  
      
    for block in partition:  
        target_blocks = set()  
        for state in block:  
            target_blocks.add(block_of[system.f[state]])  
        if len(target_blocks) > 1:  
            return False  
    return True

def kaprekar_step(n: int, base: int = 10, digits: int = 4) -> int:  
    s = str(n).zfill(digits)  
    desc = int("".join(sorted(s, reverse=True)))  
    asc = int("".join(sorted(s)))  
    return desc - asc  
  
def kaprekar_system(digits: int = 4, base: int = 10) -> FDDS:  
    max_val = base ** digits  
    transition = {}  
    for i in range(max_val):  
        transition[i] = kaprekar_step(i, base, digits)  
    return FDDS(transition)

class EvidenceObject:  
    """  
    Certificate Schema v3.1

    - Fix 4: Registry version in dependencies  
    - Fix 6: Status is UNVERIFIED until verifier runs

            raise ValueError(f"Unknown claim: {claim_id}")  
        self.claim_id = claim_id  
        self.system = system  
        self.metadata = metadata or {}  
        self.audit = get_audit()  
        self.evidence: Dict[str, Any] = {}  
        self.status = "UNVERIFIED"  # Fix 6  
        self._compute()

        if self.claim_id == "AQ-CLAIM-001":

        elif self.claim_id == "AQ-CLAIM-004":

        certificate = {

            "aqarion_version": "0.4.1-final",  
            "certificate_version": "3.1.0",

def verify_certificate(cert: Dict[str, Any]) -> Dict[str, Any]:  
    """  
    Independent verification with:  
    - Fix 4: Registry version checking  
    - Fix 5: Claimed vs recomputed separation  
    - Fix 6: Status VALID/INVALID (never from certificate file)  
    """  
    report = {  
        "claim_id": cert.get("certificate", {}).get("claim_id", "UNKNOWN"),  
        "overall_valid": False,  
        "checks": {}  
    }  
      
    certificate = cert.get("certificate", {})  
      
    # 1. Hash integrity  
    stored_hash_obj = certificate.get("hash", {})  
    stored_hash = stored_hash_obj.get("value", "") if isinstance(stored_hash_obj, dict) else stored_hash_obj  
      
    hash_payload = {  
        "claim_id": certificate.get("claim_id"),  
        "claim_statement": certificate.get("claim_statement"),  
        "status": certificate.get("status"),  
        "dependencies": certificate.get("dependencies"),  
        "system": certificate.get("system"),  
        "evidence": certificate.get("evidence"),  
        "proof_obligation": certificate.get("proof_obligation")  
    }  
    canonical = json.dumps(hash_payload, sort_keys=True, separators=(',', ':'))  
    computed_hash = hashlib.sha256(canonical.encode()).hexdigest()  
      
    report["checks"]["hash_integrity"] = {  
        "valid": computed_hash == stored_hash,  
        "stored": stored_hash[:16] + "...",  
        "computed": computed_hash[:16] + "..."  
    }  
      
    # 2. Registry version check (Fix 4)  
    deps = certificate.get("dependencies", {})  
    cert_registry = deps.get("registry_version", "UNKNOWN")  
    report["checks"]["registry_version"] = {  
        "valid": cert_registry == REGISTRY_VERSION,  
        "certificate_registry": cert_registry,  
        "current_registry": REGISTRY_VERSION  
    }  
      
    # 3. Dependency existence  
    def_deps = deps.get("definitions", [])  
    thm_deps = deps.get("theorems", [])  
    report["checks"]["dependency_existence"] = {  
        "valid": all(d in DEFINITIONS for d in def_deps) and all(t in THEOREMS for t in thm_deps),  
        "missing_definitions": [d for d in def_deps if d not in DEFINITIONS],  
        "missing_theorems": [t for t in thm_deps if t not in THEOREMS]  
    }  
      
    # 4. System reconstruction  
    sys_data = certificate.get("system", {})  
    if not sys_data or "transition" not in sys_data:  
        report["checks"]["system_reconstruction"] = {"valid": False, "error": "missing system data"}  
        report["status"] = "INVALID"  
        return report  
      
    transition = {str(k): v for k, v in sys_data["transition"].items()}  
    system = FDDS(transition)  
    report["checks"]["system_reconstruction"] = {  
        "valid": True,  
        "states": system.n,  
        "transition_hash_matches": system.transition_hash() == sys_data.get("transition_hash", "")  
    }  
      
    # 5. Claim-specific verification with claimed/recomputed separation (Fix 5)  
    claim_id = certificate.get("claim_id")  
    claimed_evidence = certificate.get("evidence", {})  
    recomputed = {}  
    agreement = {}  
      
    if claim_id == "AQ-CLAIM-001":  
        partition = compute_observable_partition(system)  
        D = defect_operator(system, partition)  
        d_zero = mat_is_zero(D)  
          
        recomputed["defect_is_zero"] = d_zero  
        recomputed["defect_norm_sq"] = str(mat_frobenius_norm_sq(D))  
        agreement["defect_is_zero"] = (d_zero == claimed_evidence.get("defect_is_zero"))  
          
        report["checks"]["claim_001_defect_zero"] = {  
            "valid": agreement["defect_is_zero"],  
            "claimed": claimed_evidence.get("defect_is_zero"),  
            "recomputed": d_zero,  
            "agreement": agreement["defect_is_zero"]  
        }  
      
    elif claim_id == "AQ-CLAIM-002":  
        cycles = find_cycles(system)  
        claimed_cycles = claimed_evidence.get("cycles", [])  
        cycles_ok = len(cycles) == len(claimed_cycles)  
        if cycles_ok:  
            cycles_ok = sorted(len(c) for c in cycles) == sorted(len(c) for c in claimed_cycles)  
          
        recomputed["cycle_count"] = len(cycles)  
        recomputed["cycle_lengths"] = [len(c) for c in cycles]  
        agreement["cycle_count"] = cycles_ok  
          
        report["checks"]["claim_002_cycles"] = {  
            "valid": cycles_ok,  
            "claimed_count": len(claimed_cycles),  
            "recomputed_count": len(cycles),  
            "agreement": cycles_ok  
        }  
      
    elif claim_id == "AQ-CLAIM-003":  
        filt = image_filtration(system)  
        claimed_filt = claimed_evidence.get("filtration", [])  
        cycles = find_cycles(system)  
        periodic = sum(len(c) for c in cycles)  
        stab_correct = (filt[-1] == periodic) if filt else False  
          
        recomputed["filtration"] = filt  
        recomputed["stabilizes_at"] = filt[-1] if filt else 0  
        recomputed["stabilization_correct"] = stab_correct  
        agreement["filtration"] = (filt == claimed_filt)  
        agreement["stabilization_correct"] = (stab_correct == claimed_evidence.get("stabilization_correct"))  
          
        report["checks"]["claim_003_filtration"] = {  
            "valid": agreement["filtration"] and agreement["stabilization_correct"],  
            "claimed_filtration": claimed_filt,  
            "recomputed_filtration": filt,  
            "claimed_stabilization": claimed_evidence.get("stabilization_correct"),  
            "recomputed_stabilization": stab_correct,  
            "agreement": agreement  
        }  
      
    elif claim_id == "AQ-CLAIM-004":  
        th = system.transition_hash()  
        recomputed["transition_hash"] = th  
        agreement["hash"] = (th == claimed_evidence.get("transition_hash"))  
          
        report["checks"]["claim_004_hash"] = {  
            "valid": agreement["hash"],  
            "claimed": claimed_evidence.get("transition_hash", "")[:16] + "...",  
            "recomputed": th[:16] + "...",  
            "agreement": agreement["hash"]  
        }  
      
    # Overall (Fix 6: status from verifier, never from file)  
    all_valid = all(c.get("valid", False) for c in report["checks"].values())  
    report["overall_valid"] = all_valid  
    report["status"] = "VALID" if all_valid else "INVALID"  
      
    return report

"""  
AQARION INDEPENDENT VERIFIER v0.4.1-final  
Dependency-aware. Claimed/recomputed separation. Status from verifier only.  
"""  
  
import json  
import sys  
from pathlib import Path  
from aqarion_kernel import verify_certificate  
  
  
def print_report(report: dict):  
    print("=" * 60)  
    print("AQARION INDEPENDENT VERIFICATION REPORT")  
    print("=" * 60)  
    print(f"Claim: {report['claim_id']}")  
    print(f"Status: {report.get('status', 'UNKNOWN')}")  
    print()  
      
    for check_name, check_result in report["checks"].items():  
        status = "PASS" if check_result.get("valid") else "FAIL"  
        print(f"{check_name:35s} {status}")  
        if "error" in check_result:  
            print(f"  Error: {check_result['error']}")  
        elif "claimed" in check_result and "recomputed" in check_result:  
            print(f"  Claimed:    {check_result['claimed']}")  
            print(f"  Recomputed: {check_result['recomputed']}")  
            if "agreement" in check_result:  
                print(f"  Agreement:  {check_result['agreement']}")  
      
    print()  
    if report.get("overall_valid"):  
        print("RESULT: CERTIFICATE VALID")  
    else:  
        print("RESULT: CERTIFICATE INVALID")  
    print("=" * 60)  
  
  
def main():  
    if len(sys.argv) < 2:  
        print("Usage: python verify_certificate.py <certificate.json>")  
        sys.exit(1)  
    cert_path = Path(sys.argv[1])  
    if not cert_path.exists():  
        print(f"Error: File not found: {cert_path}")  
        sys.exit(1)  
    with open(cert_path) as f:  
        cert = json.load(f)  
    report = verify_certificate(cert)  
    print_report(report)  
    sys.exit(0 if report.get("overall_valid") else 1)  
  
  
if __name__ == "__main__":  
    main()
