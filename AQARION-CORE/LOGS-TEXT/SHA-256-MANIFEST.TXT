import os
import hashlib

benchmark_dir = '/mnt/agents/output/benchmarks/kaprekar_gap'
manifest_lines = []

for fname in sorted(os.listdir(benchmark_dir)):
    fpath = os.path.join(benchmark_dir, fname)
    with open(fpath, 'rb') as f:
        file_hash = hashlib.sha256(f.read()).hexdigest()
    manifest_lines.append(f"{file_hash}  {fname}")

manifest_content = "# AQ-001 Kaprekar Gap Benchmark — SHA256 Manifest\n"
manifest_content += "# Generated: 2026-07-15\n"
manifest_content += "#\n"
for line in manifest_lines:
    manifest_content += line + "\n"

with open(os.path.join(benchmark_dir, 'sha256_manifest.txt'), 'w') as f:
    f.write(manifest_content)

print("sha256_manifest.txt created")
print(manifest_content)
